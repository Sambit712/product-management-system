ADMIN TABLE :

	CREATE SEQUENCE admin_seq START WITH 1;

	CREATE OR REPLACE TRIGGER admin_tgr
	BEFORE INSERT ON admin
	FOR EACH ROW
	BEGIN
		:new.id := admin_seq.NEXTVAL;
	END;
	/
